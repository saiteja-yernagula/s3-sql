deptno=(select deptno from dept where dname='sales');
                                 
b)select clause

   avg from emp;

c)from clause 
select  avg(sal) from (select ename ,sal from emp)

multi row sub query
Find employees earning more than the
lowest salary in department 30.
 any returns true if it satisfies atleast one          
 select ename from emp where sal >
     any(select sal from emp where deptno=30);

Find employees earning more than all salaries 
in department 20
 all returns true only if it satisfies all           
 select ename from emp where sal > 
    all(select sal from emp where deptno=30);

select ename from emp where 
     sal>(select avg(sal) from emp);
